cmake_minimum_required(VERSION 2.8)

project(onions-hs)

find_package(Boost COMPONENTS system REQUIRED)

#compiler options for Clang and GCC/G++
if("${CMAKE_CXX_COMPILER_ID}" STREQUAL "Clang")
  set(CMAKE_CXX_FLAGS "-g -O3 --std=c++11 -Wno-c++98-compat-pedantic -pedantic -Weverything -Wno-documentation -Wno-documentation-unknown-command -Wno-reserved-id-macro -Wno-missing-noreturn")
elseif("${CMAKE_CXX_COMPILER_ID}" STREQUAL "GNU")
  set(CMAKE_CXX_FLAGS "-g -O3 --std=c++11 -Wall -Wextra -pedantic -Wdouble-promotion -Wfloat-equal -Wunsafe-loop-optimizations")
endif()

add_executable(onions-hs
  main.cpp
  HS.cpp
)

include_directories(/usr/include/onions-common/include /usr/include/botan-1.10 /usr/include/jsoncpp)

target_link_libraries(onions-hs onions-common onions-cyoencode jsoncpp scrypt popt botan-1.10 ${Boost_SYSTEM_LIBRARY} ${Boost_LIBRARIES})

#installation of files
set(ASSETS "${CMAKE_CURRENT_SOURCE_DIR}/assets")
install(TARGETS onions-hs                DESTINATION /usr/bin/)
install(FILES ${ASSETS}/onions-hs.1.gz   DESTINATION /usr/share/man/man1)
